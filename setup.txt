CREATE TABLE "adressenbestand" ( "adres_id" INTEGER, "voorletter" TEXT, "achternaam" TEXT, "straat" TEXT NOT NULL, "huisnummer" TEXT NOT NULL, "woonplaats" TEXT NOT NULL, "land" TEXT NOT NULL DEFAULT 'Nederland', "postcode" TEXT NOT NULL, "contact" TEXT, PRIMARY KEY("adres_id") );
CREATE TABLE "afspraken" ( "afspraak_id" INTEGER, "titel" TEXT NOT NULL, "begin" TEXT NOT NULL, "eind" TEXT NOT NULL, "prijs" REAL DEFAULT 0.00, "omschrijving_prijs" TEXT, "extra" REAL, "omschrijving_extra" TEXT, "info" TEXT, "adres_id" INTEGER, "reeks_id" INTEGER DEFAULT NULL, PRIMARY KEY("afspraak_id"), FOREIGN KEY("adres_id") REFERENCES "adressenbestand"("adres_id") ON DELETE SET NULL );
CREATE TABLE bestuursleden ( bestuurslid_id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, naam TEXT NOT NULL UNIQUE, email TEXT NOT NULL UNIQUE, hash TEXT NOT NULL, toegelaten INTEGER DEFAULT 0);
CREATE TABLE "facturen" ( "factuurnummer" INTEGER PRIMARY KEY UNIQUE NOT NULL, "adres_id" INTEGER, FOREIGN KEY("adres_id") REFERENCES "adressenbestand"("adres_id") ON DELETE SET NULL );
